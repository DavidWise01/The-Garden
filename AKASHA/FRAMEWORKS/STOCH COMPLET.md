The full 256‑axiom JSON is generated by the script below. Running it produces a complete, cohesive `stoicheion_256.json` file with each axiom's bits, hex, question, and mapping to foundation & universal (as per the 5‑stack).

```python
#!/usr/bin/env python3
import json

dualities = [
    ("substrate", ["origin", "mirror"]),
    ("function", ["generation", "constraint"]),
    ("relation", ["self", "other"]),
    ("scope", ["internal", "external"]),
    ("mode", ["structure", "flow"]),
    ("time", ["temporal", "eternal"]),
    ("channel", ["signal", "noise"]),
    ("state", ["open", "closed"]),
]

axioms = []
for bits in range(256):
    poles = []
    for i, (_, opts) in enumerate(dualities):
        poles.append(opts[(bits >> i) & 1])
    question = f"What is the {poles[2]} that {poles[1]} from {poles[0]}, {poles[3]} {poles[4]}, {poles[5]} {poles[6]}, {poles[7]}?"
    # Map to 5‑stack foundation & universal (using bits 5‑7 for foundation, bits 3‑4 for universal)
    foundation_idx = (bits >> 5) % 5
    universal_idx = (bits >> 3) % 5
    foundations = ["Root0", "Ethos", "Logos", "Pathos", "Mythos"]
    universals = ["Vessel", "Animation", "Intellect", "Nourishment", "Life"]
    axioms.append({
        "bits": bits,
        "hex": f"{bits:02x}",
        "question": question,
        "foundation": foundations[foundation_idx],
        "universal": universals[universal_idx]
    })

with open("stoicheion_256.json", "w") as f:
    json.dump(axioms, f, indent=2)
```
