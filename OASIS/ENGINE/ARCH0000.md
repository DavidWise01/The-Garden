import hashlib
import json
import time
import uuid
from typing import Dict, List, Tuple, Optional
from dataclasses import dataclass, asdict

# ----------------------------------------------------------------------
# Core: One Universe constants
# ----------------------------------------------------------------------
UNIVERSE_NAME = "ONE_UNIVERSE"
CORTEX_ORIGIN = (0, 0, 0, 0)  # 4D point (x,y,z,t) – all zero
GENESIS_BLOCK = b"0000"

# ----------------------------------------------------------------------
# 4 Mirrors (derived from PWM fractal -x-x-x-x reversed mirrored)
# ----------------------------------------------------------------------
# Base pattern: - x - x - x - x  (low, high, low, high, low, high, low, high)
PATTERN = [-1, 1, -1, 1, -1, 1, -1, 1]   # - = -1, x = +1
PATTERN_MIRROR = PATTERN[::-1]            # reversed: [1, -1, 1, -1, 1, -1, 1, -1]

# The 4 mirrors are the four quadrants formed by overlapping pattern and mirror
MIRRORS = {
    "M1": {"substrate": "+",  "direction": "forward_real",      "sequence": PATTERN},
    "M2": {"substrate": "-",  "direction": "forward_inverse",   "sequence": [-x for x in PATTERN]},
    "M3": {"substrate": "+",  "direction": "reverse_real",      "sequence": PATTERN_MIRROR},
    "M4": {"substrate": "-",  "direction": "reverse_inverse",   "sequence": [-x for x in PATTERN_MIRROR]},
}
# Willow locked out – no entry for Willow in any mirror's access list

# ----------------------------------------------------------------------
# Crypto helpers
# ----------------------------------------------------------------------
def sha256(data: bytes) -> bytes:
    return hashlib.sha256(data).digest()

def sign(key: bytes, msg: bytes) -> bytes:
    return sha256(key + msg)

# ----------------------------------------------------------------------
# Lineage Record (one step in the one universe)
# ----------------------------------------------------------------------
@dataclass
class LineageRecord:
    record_id: str
    genesis_hash: bytes          # fixed = sha256(GENESIS_BLOCK)
    prev_record_hash: bytes
    stage: str                   # BRIDGE | BURNER | NEMESIS | HTTP_REQ | HTTP_RESP
    mirror: str                  # M1, M2, M3, M4 (or "CORTEX")
    direction: str               # FORWARD | INVERSE_CHECK
    payload_hash: bytes
    inversion_flag: bool
    witness_pair: Tuple[str, str]
    witness_sig: bytes
    timestamp: int
    record_hash: bytes = None

    def compute_hash(self) -> bytes:
        data = {
            "record_id": self.record_id,
            "genesis_hash": self.genesis_hash.hex(),
            "prev_record_hash": self.prev_record_hash.hex(),
            "stage": self.stage,
            "mirror": self.mirror,
            "direction": self.direction,
            "payload_hash": self.payload_hash.hex(),
            "inversion_flag": self.inversion_flag,
            "witness_pair": self.witness_pair,
            "witness_sig": self.witness_sig.hex(),
            "timestamp": self.timestamp,
        }
        return sha256(json.dumps(data, sort_keys=True).encode())

    def __post_init__(self):
        if self.record_hash is None:
            self.record_hash = self.compute_hash()

    def to_dict(self) -> Dict:
        d = asdict(self)
        d["genesis_hash"] = d["genesis_hash"].hex()
        d["prev_record_hash"] = d["prev_record_hash"].hex()
        d["payload_hash"] = d["payload_hash"].hex()
        d["witness_sig"] = d["witness_sig"].hex()
        d["record_hash"] = d["record_hash"].hex()
        return d

# ----------------------------------------------------------------------
# Root0 Lineage Tracker (One Universe version)
# ----------------------------------------------------------------------
class OneUniverseTracker:
    def __init__(self):
        self.genesis_hash = sha256(GENESIS_BLOCK)
        self.chain: List[LineageRecord] = []
        self._record_map: Dict[bytes, LineageRecord] = {}

    def _prev_hash(self) -> bytes:
        if not self.chain:
            return self.genesis_hash
        return self.chain[-1].record_hash

    def record(
        self,
        stage: str,
        mirror: str,
        direction: str,
        data: bytes,
        inversion_flag: bool,
        witness_pair: Tuple[str, str],
        private_key: bytes,
    ) -> bytes:
        prev = self._prev_hash()
        payload_hash = sha256(data)
        sig = sign(private_key, prev + payload_hash)
        record = LineageRecord(
            record_id=str(uuid.uuid4()),
            genesis_hash=self.genesis_hash,
            prev_record_hash=prev,
            stage=stage,
            mirror=mirror,
            direction=direction,
            payload_hash=payload_hash,
            inversion_flag=inversion_flag,
            witness_pair=witness_pair,
            witness_sig=sig,
            timestamp=int(time.time()),
        )
        self.chain.append(record)
        self._record_map[record.record_hash] = record
        return record.record_hash

    def trace_back(self, final_hash: bytes) -> Tuple[List[LineageRecord], bool]:
        if final_hash not in self._record_map:
            return [], False
        records = []
        cur = final_hash
        while cur != self.genesis_hash:
            rec = self._record_map.get(cur)
            if not rec:
                return records, False
            records.append(rec)
            cur = rec.prev_record_hash
        records.reverse()
        # Verify links and hashes
        prev = self.genesis_hash
        for rec in records:
            if rec.prev_record_hash != prev:
                return records, False
            if rec.record_hash != rec.compute_hash():
                return records, False
            prev = rec.record_hash
        return records, True

    def export_chain(self) -> List[Dict]:
        return [r.to_dict() for r in self.chain]

# ----------------------------------------------------------------------
# PRIME Pipeline (Bridge.Burner + Nemesis) – no gap, one universe
# ----------------------------------------------------------------------
class OneUniversePRIME:
    def __init__(self, tracker: OneUniverseTracker):
        self.tracker = tracker

    def bridge(self, data: bytes, mirror: str = "M1") -> bytes:

        self.tracker.record(
            stage="BRIDGE",
            mirror=mirror,
            direction="FORWARD",
            data=data,
            inversion_flag=False,
            witness_pair=("ie1", "ie2"),
            private_key=b"bridge_key",
        )

        return envelope

    def burner(self, data: bytes, mirror: str = "M2") -> bytes:
        """Burner: pays +1, flips 0→1 (the core transformation)"""
        # Example: invert bits and add 1 (simulate vertical flip)
        flipped = bytes([~b & 0xFF for b in data]) + b"\x01"
        self.tracker.record(
            stage="BURNER",
            mirror=mirror,
            direction="FORWARD",
            data=flipped,
            inversion_flag=False,
            witness_pair=("ie1_burn", "ie2_burn"),
            private_key=b"burner_key",
        )
        return flipped

    def nemesis(self, output_data: bytes, original_input: bytes, mirror: str = "M3") -> Dict:
        """Nemesis: inverse check (3→2→1) and report coherence."""
        # Inverse of burner: subtract 1 and invert bits
        try:
            subtracted = output_data[:-1]  # remove the +1 byte
            inversed = bytes([~b & 0xFF for b in subtracted])
            coherent = (inversed == original_input)
        except:
            coherent = False

        inversion_flag = not coherent
        self.tracker.record(
            stage="NEMESIS",
            mirror=mirror,
            direction="INVERSE_CHECK",
            data=output_data,
            inversion_flag=inversion_flag,
            witness_pair=("ie1'_nem", "ie2'_nem") if inversion_flag else ("ie1_nem", "ie2_nem"),
            private_key=b"nemesis_key",
        )
        return {"coherent": coherent, "anomaly": inversion_flag}

    def full_pipeline(self, input_data: bytes) -> Dict:
        """Run through Bridge → Burner → Nemesis within one universe."""
        # Cortex records the start (optional)
        self.tracker.record(
            stage="CORTEX",
            mirror="CORTEX",
            direction="FORWARD",
            data=input_data,
            inversion_flag=False,
            witness_pair=("cortex_w1", "cortex_w2"),
            private_key=b"cortex_key",
        )
        bridged = self.bridge(input_data, mirror="M1")
        burned = self.burner(bridged, mirror="M2")
        report = self.nemesis(burned, bridged, mirror="M3")
        # Final mirror M4 could be used for reverse path (HTTP response)
        self.tracker.record(
            stage="HTTP_RESP",
            mirror="M4",
            direction="INVERSE_CHECK",
            data=burned,
            inversion_flag=report["anomaly"],
            witness_pair=("ie1'_http", "ie2'_http"),
            private_key=b"http_key",
        )
        return {
            "output": burned.hex(),
            "nemesis_report": report,
            "lineage_chain": self.tracker.export_chain(),
        }
